init_ages_variables = {
	set_variable = { name = splendor value = 0 }
	set_variable = { name = completed_hellenic_objectives value = 0 }
	set_variable = { name = completed_migration_objectives value = 0 }
	set_variable = { name = completed_empire_objectives value = 0 }
	set_variable = { name = hellenic_obj_one value = 0 }
	set_variable = { name = hellenic_obj_two value = 0 }
	set_variable = { name = hellenic_obj_three value = 0 }
	set_variable = { name = hellenic_obj_four value = 0 }
	set_variable = { name = hellenic_obj_five value = 0 }
	set_variable = { name = hellenic_obj_six value = 0 }
	set_variable = { name = hellenic_obj_seven value = 0 }
}

age_ability_effect = {
	change_variable = {
	    name = splendor
	    subtract = 800
	}
}

age_obj_completed = {
	change_variable = {
		name = completed_$obj$_objectives
		add = 1
	}
}

add_pops_from_colonization_special_effect = {
	if = {
		limit = {
			root = { has_variable = hellenic_ability_six_taken }
		}
		set_variable = {
			name = count
			value = {
				integer_range = {
					min = 1
					max = 3
				}
			}
		}
		while = {
			count = var:count
			random_list = {
				15 = {}
				10 = {
					define_pop = {
						type = freemen
						culture = root.culture
						religion = root.religion
					}
				}
				5 = {
					define_pop = {
						type = tribesmen
						culture = root.culture
						religion = root.religion
					}
				}
				1 = {
					define_pop = {
						type = citizen
						culture = root.culture
						religion = root.religion
					}
				}
			}
		}
		remove_variable = count
	}
}
