add_pops_from_colonization_special_effect = {
	if = {
		limit = {
			root = { has_variable = hellenic_ability_six_taken }
		}
		set_variable = {
			name = count
			value = {
				integer_range = {
					min = 1
					max = 3
				}
			}
		}
		while = {
			count = var:count
			random_list = {
				15 = {}
				10 = {
					define_pop = {
						type = freemen
						culture = root.culture
						religion = root.religion
					}
				}
				5 = {
					define_pop = {
						type = tribesmen
						culture = root.culture
						religion = root.religion
					}
				}
				1 = {
					define_pop = {
						type = citizen
						culture = root.culture
						religion = root.religion
					}
				}
			}
		}
		remove_variable = count
	}
}

migration_age_looting_effect = {
	if = {
		limit = {
			is_age = { num = 2 }
			local_var:sieged_city = {
				total_population >= 20
			}
			var:migration_obj_two < 100
		}
		change_variable = {
			name = migration_obj_two
			add = 20
		}
	}
}
