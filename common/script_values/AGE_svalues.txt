hellenic_obj_four_svalue = {
	value = 0
	capital_scope = {
		if = {
			limit = {
				state = { incoming_trade_routes >= 10 }
			}
			add = 1
		}
		if = {
			limit = {
				state = { outgoing_trade_routes >= 10 }
			}
			add = 1
		}
		if = {
			limit = {
				any_neighbor_province = {
					count = all
					is_sea = no
					is_uninhabitable = no
					has_road_towards = PREV
				}
			}
			add = 1
		}
	}
}
