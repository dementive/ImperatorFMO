char_can_royal_marry_trigger = {
    scope = character
    is_shown = {
        age > 10
        is_ruler = no
        is_married = no
    }
}

remove_bloodline_list_variables_button = {
    scope = country
    effect = {
        every_in_list = {
            variable = player_bloodline_list
            root = {
                remove_list_variable = {
                    name = player_bloodline_list
                    target = prev
                }
            }
        }
    }
}
