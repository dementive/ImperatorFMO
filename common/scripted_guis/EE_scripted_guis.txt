show_governorship_states_button = {
	scope = governorship
	saved_scopes = {
		player
	}
	ai_is_valid = {
		always = no
	}
	effect = {
		every_governorship_state = {
			scope:player = {
				add_to_variable_list = {
					name = player_governorship_states
					target = prev
				}
			}
		}
	}
}
show_trade_centers_in_governorship = {
	scope = governorship
	saved_scopes = {
		player
	}
	ai_is_valid = {
		always = no
	}
	effect = {
		capital_scope = {
			region = {
				every_region_province = {
					limit = {
						OR = {
							has_province_modifier = regional_center_of_trade_level_1
							has_province_modifier = regional_center_of_trade_level_2
							has_province_modifier = regional_center_of_trade_level_3
							has_province_modifier = regional_center_of_trade_level_4
							has_province_modifier = regional_center_of_trade_level_5
						}
					}
					scope:player = {
						add_to_variable_list = {
							name = region_trade_centers
							target = prev
						}
					}
				}
			}
		}
	}
}

upgrade_trade_center_button = {
	scope = province
	saved_scopes = {
		player
	}
	ai_is_valid = {
		always = no
	}
	is_shown = {
		#owner = scope:player
	}
	is_valid = {
		trigger_if = {
			limit = {
				NOT = {
					owner = scope:player
				}
			}
			owner = scope:player
		}
		trigger_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_5
			}
			custom_tooltip = {
				text = max_level_trade_center_tt
				NOT = {
					has_province_modifier = regional_center_of_trade_level_5
				}
			}
		}
		trigger_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_1
			}
			hidden:owner = {
				treasury > 200
				political_influence > 50
			}
		}
		trigger_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_2
			}
			hidden:owner = {
				treasury > 500
				political_influence > 75
			}
		}
		trigger_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_3
			}
			hidden:owner = {
				treasury > 1000
				political_influence > 100
			}
		}
		trigger_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_4
			}
			hidden:owner = {
				treasury > 2500
				political_influence > 150
			}
		}
		trigger_if = {
			limit = {
				has_province_modifier = upgrading_trade_center
			}
			NOT = { has_province_modifier = upgrading_trade_center }
		}
	}
	effect = {
		if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_1
			}
			owner = {
				add_political_influence = -50
				add_treasury = -200
			}
			custom_tooltip = after_2_years_tooltip
			show_as_tooltip = {
				add_province_modifier = {
					name = regional_center_of_trade_level_2
					duration = -1
				}
			}
			hidden_effect = { 
				add_province_modifier = {
					name = upgrading_trade_center
					duration = 730
				}
				trigger_event = {
					id = ee_economy_events.5
					days = 730
				}
			}
		}
		else_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_2
			}
			owner = {
				add_political_influence = -75
				add_treasury = -500
			}
			custom_tooltip = after_2_years_tooltip
			show_as_tooltip = {
				add_province_modifier = {
					name = regional_center_of_trade_level_3
					duration = -1
				}
			}
			hidden_effect = { 
				add_province_modifier = {
					name = upgrading_trade_center
					duration = 730
				}
				trigger_event = {
					id = ee_economy_events.6
					days = 730
				}
			}
		}
		else_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_3
			}
			owner = {
				add_political_influence = -100
				add_treasury = -1000
			}
			custom_tooltip = after_2_years_tooltip
			show_as_tooltip = {
				add_province_modifier = {
					name = regional_center_of_trade_level_4
					duration = -1
				}
			}
			hidden_effect = { 
				add_province_modifier = {
					name = upgrading_trade_center
					duration = 730
				}
				trigger_event = {
					id = ee_economy_events.7
					days = 730
				}
			}
		}
		else_if = {
			limit = {
				has_province_modifier = regional_center_of_trade_level_4
			}
			owner = {
				add_political_influence = -150
				add_treasury = -2500
			}
			custom_tooltip = after_2_years_tooltip
			show_as_tooltip = {
				add_province_modifier = {
					name = regional_center_of_trade_level_4
					duration = -1
				}
			}
			hidden_effect = { 
				add_province_modifier = {
					name = upgrading_trade_center
					duration = 730
				}
				trigger_event = {
					id = ee_economy_events.8
					days = 730
				}
			}
		}
	}
}

set_grain_focus = {
    scope = country
    confirm_title = "grain_focus_title"
    confirm_text = "grain_focus_desc"
    ai_is_valid = {
        always = no
    }
    is_valid = {
        trigger_if = {
            limit = {
                political_influence < 50
            }
            political_influence >= 50
        }
        trigger_if = {
            limit = {
                NOT = {
                    any_owned_province = {
                        trade_goods = grain
                    }
                }
            }
            any_owned_province = {
                trade_goods = grain
            }
        }
        trigger_if = {
            limit = {
                has_variable = grain_focus
            }
            custom_tooltip = {
                text = this_is_already_the_trade_good_focus_tt
                NOT = { has_variable = grain_focus }
            }
        }
    }
    effect = {
        remove_trade_good_focus_variables_effect = yes
        set_variable = grain_focus
        add_political_influence = -50
        custom_tooltip = all_grain_provinces_gain
        random_owned_province = {
            show_as_tooltip = {
                add_province_modifier = {
                    name = production_focus
                    duration = -1
                }
            }
        }
        hidden_effect = {
            every_owned_province = {
                limit = {
                    has_province_modifier = production_focus
                }
                remove_province_modifier = production_focus
            }
            every_owned_province = {
                limit = { trade_goods = grain }
                add_province_modifier = {
                    name = production_focus
                    duration = -1
                }
            }
        }
    }
}
